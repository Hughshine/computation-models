\problem{证明二元不动点定理：对于任意$F,G\in\Lambda$，存在$X, Y\in\Lambda$，满足  $FXY=X, GXY=Y$.
% $$ 
% \begin{aligned}
% & FXY=X,\\    
% & GXY=Y. \\
% \end{aligned}
% $$
}
\begin{pf} \rm 
    设解向量$A=[X, Y]$，\\
    等式组等价于等式$(\lambda z.[F(z)_1^2 (z)_2^2, G(z)_1^2 (z)_2^2]) A = A$. 等式组的解等价于求该等式的解. 由一元不动点定理可知该等式存在不动点$A$，所以等式组也存在解$X, Y$.
    \qed
\end{pf}
